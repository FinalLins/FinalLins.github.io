1. 概述

  前段时间我们学习了堆排序，那么我们今天就来彻底的学习堆这个数据结构。数据结构堆跟堆内存是不一样的。数据结构堆本质其实一种完全二叉树。（树的知识点这里就跳过了，不了解的同学可以先去看看）。

  • 堆是一个近似完全二叉树的结构：除了最后一层节点可以不是满的，其他的节点从左到右都是满的。
  • 堆满足堆积的性质：即子结点的键值总是小于等于（或者大于等于）它的父节点。

结合二叉树的性质计算出索引关系，父结点 (i - 1) / 2 ，左子结点 2 * i + 1 ，右子结点 2 * i + 2。
最大堆：结点的值大于等于子结点得值，key[i] >= key[2 * i + 1] 并且 key[i] >= key[2 * i + 2]。
最小堆：结点的值小于等于子结点的值，key[i] <= key[2 * i + 1] 并且 key[i] <= key[2 * i + 2]。

2. 代码实现

class MyHeap {
    /**
     * 节点数组
     */
    private int[] array;

    /**
     * 容量
     */
    private int capacity;

    /**
     * 当前大小
     */
    private int size;

    public MyHeap(int capacity) {
        this.array = new int[capacity];
        this.capacity = capacity;
        this.size = 0;
    }

    public boolean isEmpty() {
        return size == 0;
    }

    public boolean isFull() {
        return size == capacity;
    }

    /**
     * 插入一个元素
     *
     * @param element
     * @return
     */
    public boolean add(int element) {
        if (isFull()) {                     // 存储已满
            return false;
        }
        array[size] = element;              // 将新节点放到数组最后
        adjustUp(size++);                   // 调整堆结构
        return true;
    }

    /**
     * 向上调整
     * 1. 将待插入元素放到数组最后；
     * 2. 令当前节点为插入元素所在节点；
     * 3. 当前节点与父节点比较大小，如果父节点的关键字小于当前节点，则与父节点交换；否则，表示调整结构结束，退出；
     * 4. 如果没有退出，对交换后的当前节点重复上一步过程。
     */
    private void adjustUp(int index) {
        int parent = (index - 1) / 2;       //父节点的索引
        int last = array[index];            //将新加的尾节点存在last中
        while (index > 0 && array[parent] < last) {
            array[index] = array[parent];
            index = parent;
            parent = (parent - 1) / 2;
        }
        array[index] = last;
    }

    /**
     * 堆中删除元素，总是删除堆顶元素
     *
     * @return
     */
    public int remove() {
        if (isEmpty()) {
            throw new RuntimeException("堆为空，不能进行删除操作");
        }

        int first = array[0];
        array[0] = array[--size];           //最后一个元素放到堆顶
        adjustDown(0);
        return first;
    }

    /**
     * 向下调整
     * 1. 删除堆顶元素，并将最后一个元素放到堆顶；
     * 2. 设当前节点为堆顶节点；
     * 3. 找到当前节点的子节点中关键字最大的那个节点，然后将此子节点与当前节点比较大小，如果此节点关键字大于当前节点，则交换两个节点；否则，表示调整结构结束，退出；
     * 4. 如果没有退出，对交换后的当前节点重复上一步过程。
     */
    private void adjustDown(int index) {
        int first = array[index];
        int left, right, maxChild;
        while (index < size / 2) {          //后面没有子节点了就结束循环
            left = 2 * index + 1;
            right = left + 1;               //2 * index + 2;

            // maxChild = index;
            // if (left < size && array[left] > array[maxChild]) {
            //     maxChild = left;
            // }
            // if (right < size && array[right] > array[maxChild]) {
            //     maxChild = right;
            // }
            // if (maxChild == index) {
            //     break;
            // }
            // 判断右边的子节点是否存在并且右边子节点大于左边子节点，
            if (right < size && array[right] > array[left]) {
                maxChild = right;
            } else {
                maxChild = left;
            }

            if (first >= array[maxChild]) {
                break;                      //结点比子结点值都大，退出循环
            }
            array[index] = array[maxChild];
            index = maxChild;
        }
        array[index] = first;
    }

    public boolean modifyByIndex(int index, int newElement) {
        if (index < 0 || index >= size) {
            return false;
        }
        int oldElement = array[index];
        if (oldElement < newElement) {
            adjustUp(index);
        } else if (oldElement > newElement) {
            adjustDown(index);
        }
        return true;
    }

}

            myHeap.add(array[i]);
            myHeap.print();
